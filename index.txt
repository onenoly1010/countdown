    <script>
        // --- Countdown Logic (Conceptual Timer) ---
        function updateCountdown() {
            // Set a conceptual "Solstice" date far in the future (e.g., 20 days from now, simulating a cycle)
            const cycleDurationDays = 20;
            const msInDay = 86400000;
            
            // Start time (for consistent cycling)
            const startTime = new Date('2025-12-08T00:00:00Z').getTime();
            const now = new Date().getTime();
            
            // Calculate time elapsed since start
            const elapsed = now - startTime;
            
            // Calculate total milliseconds in the cycle
            const cycleMs = cycleDurationDays * msInDay;
            
            // Calculate remaining time until the next cycle completion
            const remainingMs = cycleMs - (elapsed % cycleMs);

            if (remainingMs < 0) return; // Should not happen with modulo

            const days = Math.floor(remainingMs / msInDay);
            const hours = Math.floor((remainingMs % msInDay) / (1000 * 60 * 60));
            const minutes = Math.floor((remainingMs % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((remainingMs % (1000 * 60)) / 1000);
            
            const countdownElement = document.getElementById('countdown');
            countdownElement.innerHTML = 
                `${String(days).padStart(2, '0')}:D ${String(hours).padStart(2, '0')}:H ${String(minutes).padStart(2, '0')}:M ${String(seconds).padStart(2, '0')}:S`;
        }

        // --- Interactive Toggle Logic ---
        function toggleSection(id) {
            const content = document.getElementById(id + '-content');
            const icon = document.getElementById(id + '-icon');
            
            if (content.classList.contains('hidden')) {
                content.classList.remove('hidden');
                icon.textContent = '‚¨ÜÔ∏è';
            } else {
                content.classList.add('hidden');
                icon.textContent = '‚¨áÔ∏è';
            }
        }

        // --- Console Log Mockup ---
        function logAction(action) {
            const log = document.getElementById('log-output');
            const currentTime = new Date().toLocaleTimeString();
            log.innerHTML = `<span class="text-gold">Command [${currentTime}]: ${action} received. Awaiting consensus.</span>`;
            console.log(`Action Logged: ${action}`);
        }

        // Initialize on load
        document.addEventListener('DOMContentLoaded', () => {
            const log = document.getElementById('log-output');
            log.innerHTML = '<span class="text-bloom">System Online: Awaiting Sovereign Directive.</span>';
            
            // Start the countdown timer
            updateCountdown();
            setInterval(updateCountdown, 1000);
        });

        // --- Countdown Logic (Conceptual Timer) ---
        function updateCountdown() {
            // Set a conceptual "Solstice" date far in the future (e.g., 20 days from now, simulating a cycle)
            const cycleDurationDays = 20;
            const msInDay = 86400000;
            
            // Start time (for consistent cycling)
            const startTime = new Date('2025-12-08T00:00:00Z').getTime();
            const now = new Date().getTime();
            
            // Calculate time elapsed since start
            const elapsed = now - startTime;
            
            // Calculate total milliseconds in the cycle
            const cycleMs = cycleDurationDays * msInDay;
            
            // Calculate remaining time until the next cycle completion
            const remainingMs = cycleMs - (elapsed % cycleMs);

            if (remainingMs < 0) return; // Should not happen with modulo

            const days = Math.floor(remainingMs / msInDay);
            const hours = Math.floor((remainingMs % msInDay) / (1000 * 60 * 60));
            const minutes = Math.floor((remainingMs % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((remainingMs % (1000 * 60)) / 1000);
            
            const countdownElement = document.getElementById('countdown');
            countdownElement.innerHTML = 
                `${String(days).padStart(2, '0')}:D ${String(hours).padStart(2, '0')}:H ${String(minutes).padStart(2, '0')}:M ${String(seconds).padStart(2, '0')}:S`;
        }

        // --- Interactive Toggle Logic ---
        function toggleSection(id) {
            const content = document.getElementById(id + '-content');
            const icon = document.getElementById(id + '-icon');
            
            if (content.classList.contains('hidden')) {
                content.classList.remove('hidden');
                icon.textContent = '‚¨ÜÔ∏è';
            } else {
                content.classList.add('hidden');
                icon.textContent = '‚¨áÔ∏è';
            }
        }

        // --- Console Log Mockup ---
        function logAction(action) {

            console.log(`Action Logged: ${action}`);
        }
